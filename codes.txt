# In PowerShell run this to create the zip and the hash!

cd "C:\Users\lua4006\Desktop\GitRepo\arduino-esp32-tinzr"

# 1) Prepare a clean packaging root folder
$PKGROOT = "pkgroot\TinZr-esp32-1.0.0"
Remove-Item -Recurse -Force "pkgroot" -ErrorAction SilentlyContinue
New-Item -ItemType Directory -Force -Path $PKGROOT | Out-Null

# 2) Copy the *contents* of hardware\esp32\1.0.0 into the root folder
Copy-Item -Recurse -Force "hardware\esp32\1.0.0\*" $PKGROOT

# 3) Create the ZIP with exactly one top-level folder (TinZr-esp32-1.0.0)
Remove-Item ".\tinzr-esp32-1.0.0.zip" -Force -ErrorAction SilentlyContinue
Compress-Archive -Path "pkgroot\TinZr-esp32-1.0.0" -DestinationPath "tinzr-esp32-1.0.0.zip"

# 4) Show checksum and size
Get-FileHash .\tinzr-esp32-1.0.0.zip -Algorithm SHA256
(Get-Item .\tinzr-esp32-1.0.0.zip).Length
